<<//syntax//>>
1. 링크 ( <a href = " 링크주소 " target = "_blank"> ~ </a> title = " 이름 ")
<a> ~ </a> : 글이나 문장에 링크를 만들어준다. anchor
href = " 링크주소 " : <a>만 쓰면 어느 링크로 가라고 하는지 몰라서 "링크 주소"에 링크 주소를 넣어준다. 
target = _blank : 글이나 문장을 눌러 링크를 들어갈 때 새창으로 들어가고 싶으면 사용한다.
title = " 이름 " : 커서가 링크에 닿았을 때 표시되는 이름.
